273 void TerrainGenerationAnimation( TransCache &obj ) {
274 
275  typedef enum {
276  SHRINKING, TRANSITIONING, GROWING, DONE
277  } AnimationState;
278  static AnimationState terrainState = DONE;
279  static float CurrentScale = 0.0;
280 
281  if ( switchingTerrain ) {
282  terrainState = SHRINKING;
283  switchingTerrain = false;
284  CurrentScale = 0.0;
285  }
286 
287  switch ( terrainState ) {
288 
289  case SHRINKING:
290  obj._scale.set( 1.0,
291  ((1.0 + cos( CurrentScale * DEGREES_TO_RADIANS )) / 2.0),
292  1.0 );
293 
294  CurrentScale += 1.0 * tick.scale();
295 
296  if ( CurrentScale >= 180.0 ) {
297  terrainState = TRANSITIONING;
298  }
299  break;
300 
301  case TRANSITIONING:
302  // Multithread this for graphics 2 project
303  randomize_terrain();
304  terrainState = GROWING;
305  break;
306 
307  case GROWING:
308  obj._scale.set( 1.0,
309  ((1.0 + cos( CurrentScale * DEGREES_TO_RADIANS )) / 2.0),
310  1.0 );
311 
312  CurrentScale += 1.0 * tick.scale();
313 
314  if ( CurrentScale >= 360.0 ) terrainState = DONE;
315  break;
316 
317  case DONE:
318  return;
319 
320  default:
321  return;
322  }
323 }
