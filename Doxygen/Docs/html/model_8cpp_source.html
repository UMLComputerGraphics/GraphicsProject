400  const int S = pow( 2, N ) + 1;
401  std::vector< Point4 > &vec = obj->_vertices;

403  //std::vector< vec3 > &nor = obj->_normals;

406 
407  if ( DEBUG ) printf( "\nEntering landGen()...\n" );
408  // the range (-h -> h) for the average offset
409  // This determines the jaggedness of the peaks and valleys.
410  // A smaller initial value will create smaller peaks and shallow valleys for
411  // a look more similar to rolling hill tops.
412  double CH = 0.5;
413  double magnitude = (H * (2 - pow( 2, -(N) )));
414  if ( DEBUG )
415  fprintf( stderr, "landGen theoretical magnitude: %f\n", magnitude );

418  if ( vec.size() ) vec.clear();

422  vec.push_back( vec4( j - (S / 2), 0, i - (S / 2), 1 ) );
423 
424  /* Initialize our color vectors. */
425  if ( col.size() ) col.clear();
426  col.reserve( S * S );
427  for ( int i = 0; i < S; ++i )
428  for ( int j = 0; j < S; ++j )

434  for ( int x = 0; x < S; ++x )
435  // Set coordinates so that the texture applied to
436  // smaller model coords.
437  // Prevent texture from being stretched to fit model.
438  txy.push_back(
439  vec2( fmod( 100 * (float) x / (float) S, 100 ),
440  fmod( 100 * (float) z / (float) S, 100 ) ) );
441

443  HEIGHT_AT( 0, 0 ) = 0;
444  HEIGHT_AT( S-1, 0 ) = 0;
445  HEIGHT_AT( 0, S-1 ) = 0;
446  HEIGHT_AT( S-1, S-1 ) = 0;

449  // Diamond-Square algorithm
450  // sideLength is the distance of a single square side or
451  // distance of diagonal in diamond.

460  + HEIGHT_AT( x, z + sideLength )
461  + HEIGHT_AT( x + sideLength, z + sideLength ))

465  + COLOR_AT(x,z+sideLength)
466  + COLOR_AT(x+sideLength,z+sideLength))

471  COLOR_AT( x + halfSide, z + halfSide ) = color_avg + color_jitter;
472  } // for z
473  } // for x

476  // Since diamonds are staggered, we only move x by half side
477  // NOTE: If the data shouldn't wrap the x < SIZE and y < SIZE
478  for ( int x = 0; x < S - 1; x += halfSide ) {
479  for ( int z = (x + halfSide) % sideLength; z < S - 1; z += sideLength ) {
480

483  // so that we can wrap around the array to find the corners

486  + HEIGHT_AT( x + halfSide % S, z )

488  + HEIGHT_AT( x, (z - halfSide + S) % S ))

492  + COLOR_AT( x + halfSide % S, z )
493  + COLOR_AT( x, z + halfSide % S )
494  + COLOR_AT( x, (z - halfSide + S) % S ))

499  COLOR_AT( x, z ) = color_avg + color_jitter;

503  HEIGHT_AT( S-1, z ) = HEIGHT_AT( x, z );
504  COLOR_AT( S-1, z ) = COLOR_AT( x, z );

507  HEIGHT_AT( x, S-1 ) = HEIGHT_AT( x, z );
508  COLOR_AT( x, S-1 ) = COLOR_AT( x, z );

511  } // for x

516  // Size of 1D array is 2*SIZE^2 - 2SIZE for
517  // the sake of proper parsing by glTriangleStrips

531  drawIndex.push_back( OFFSET_AT(i+1,j-1) );
532  }
533  }

539  */
