181 void wiiHandleBB( CWiimote &wm ) {
182  enum bb_sensor {
183  TOP_LEFT, TOP_RIGHT, BOT_LEFT, BOT_RIGHT, TOT_WEIGHT
184  };
185 
186  static bool tare = false;
187  static float tare_val[5] = { 0, 0, 0, 0, 0 };
188  static float raw_val[5] = { 0, 0, 0, 0, 0 };
189  static float adj_val[5] = { 0, 0, 0, 0, 0 };
190  static size_t tare_polls = 0;
191 
192  if ( wm.Buttons.isPressed( CButtons::BUTTON_A ) ) {
193  std::cerr << "Enabling Tare\n";
194  tare = true;
195  tare_polls = 0;
196  for ( size_t i = 0; i < 5; ++i )
197  tare_val[i] = 0;
198  pollResults.bbMagnitudes = vec3( 0, 0, 0 );
199  }
200  CBalanceBoard &bb = wm.ExpansionDevice.BalanceBoard;
201  bb.WeightSensor.GetWeight( raw_val[TOT_WEIGHT], raw_val[TOP_LEFT],
202  raw_val[TOP_RIGHT], raw_val[BOT_LEFT],
203  raw_val[BOT_RIGHT] );
204 
205  /* Compute our adjusted values. */
206  adj_val[TOT_WEIGHT] = 0;
207  for ( size_t i = 0; i < 4; ++i ) {
208  adj_val[i] = raw_val[i] - tare_val[i];
209  adj_val[TOT_WEIGHT] += adj_val[i];
210  }
211 
212  /* We are now either going to zero the scale,
213  or ignore the inputs if they are too small. */
214  if ( tare ) {
215  for ( size_t i = 0; i < 4; ++i )
216  tare_val[i] += raw_val[i];
217  if ( ++tare_polls >= 1000 ) {
218  std::cerr << "Tare complete.\n";
219  tare_val[4] = 0;
220  tare_polls = 0;
221  tare = false;
222  for ( size_t i = 0; i < 4; ++i ) {
223  tare_val[i] = tare_val[i] / 1000;
224  std::cerr << "Tare[" << i << "] == " << tare_val[i] << ".\n";
225  tare_val[4] += tare_val[i];
226  }
227  }
228  return; /* Return early: do not compute anything with weird half-tared values. */
229  } else if ( adj_val[TOT_WEIGHT] < 10 ) {
230  pollResults.bbMagnitudes = vec3( 0, 0, 0 );
231  return;
232  }
233 
234  /* Compute X and Y magnitudes. */
235  float surge_pct = (adj_val[TOP_LEFT] + adj_val[TOP_RIGHT] - adj_val[BOT_LEFT]
236  - adj_val[BOT_RIGHT])
237  / adj_val[TOT_WEIGHT];
238  float sway_pct = (adj_val[TOP_RIGHT] + adj_val[BOT_RIGHT] - adj_val[TOP_LEFT]
239  - adj_val[BOT_LEFT])
240  / adj_val[TOT_WEIGHT];
241 
242  /* Even with zeroing the scale and ignoring small weights,
243  Sometimes the sensors report something weird.
244  With this we ensure that we've normalized to the [-1,1] range. */
245  if ( surge_pct < -1 ) surge_pct = -1;
246  else if ( surge_pct > 1 ) surge_pct = 1;
247  if ( sway_pct < -1 ) sway_pct = -1;
248  else if ( sway_pct > 1 ) sway_pct = 1;
249 
250  pollResults.bbMagnitudes.y = sway_pct;
251  pollResults.bbMagnitudes.z = surge_pct;
252 
253  if ( 0 ) {
254  printf( "Balance Board Raw Weights: {" );
255  for ( size_t i = 0; i < 5; ++i )
256  printf( "%6f, ", raw_val[i] );
257  printf( "}\n" );
258  printf( "Balance Board Adj Weights: {" );
259  for ( size_t i = 0; i < 5; ++i )
260  printf( "%6f, ", adj_val[i] );
261  printf( "}\n" );
262  printf( "SURGE: %6f; SWAY: %6f\n", surge_pct, sway_pct );
263  }

289 void pollWii( CWii &wii, bool CalibrateGyro ) {
290 
291  struct timeval start, end;
292  pollResults.resetCamera = false;
293 
294  //Poll the wiimotes to get the status like pitch or roll
295  if ( 0 ) gettimeofday( &start, NULL );
296  if ( wii.Poll() ) {
297 
298  std::vector< CWiimote >& wiimotes = wii.GetWiimotes();
299  std::vector< CWiimote >::iterator it;
300 
301  for ( it = wiimotes.begin(); it != wiimotes.end(); ++it ) {
302 
303  // Use a reference to make working with the iterator handy.
304  CWiimote& wiimote = *it;
305 
306  switch ( wiimote.GetEvent() ) {
307 
308  case CWiimote::EVENT_EVENT:
309  if ( wiimote.ExpansionDevice.GetType()
310  == wiimote.ExpansionDevice.TYPE_BALANCE_BOARD )
311  wiiHandleBB( wiimote );
312  HandleEvent( wiimote );
313  break;
314 
315  case CWiimote::EVENT_CONNECT:
316  std::cerr << "Device connecting.\n";
317  break;
318  case CWiimote::EVENT_UNEXPECTED_DISCONNECT:
319  std::cerr << "Unexpected disconnect.\n";
320  break;
321  case CWiimote::EVENT_NUNCHUK_INSERTED:
322  enableRemote( wiimote );
323  std::cerr << "Nunchuk has become available.\n";
324  break;
325  case CWiimote::EVENT_NUNCHUK_REMOVED:
326  std::cerr << "Nunchuk has been removed.\n";
327  break;
328  case CWiimote::EVENT_BALANCE_BOARD_INSERTED:
329  std::cerr << "A balance board has become available.\n";
330  wiimote.SetLEDs( ::ledMap[3] );
331  ;
332  wiimote.SetLEDs( ::ledMap[3] );
333  ;
334  break;
335  case CWiimote::EVENT_BALANCE_BOARD_REMOVED:
336  std::cerr << "A balance board has been removed.\n";
337  break;
338 
339  case CWiimote::EVENT_CLASSIC_CTRL_INSERTED:
340  case CWiimote::EVENT_MOTION_PLUS_INSERTED:
341  enableRemote( wiimote );
342  std::cerr << "A peripheral has been attached.\n";
343  break;
344 
345  case CWiimote::EVENT_CLASSIC_CTRL_REMOVED:
346  case CWiimote::EVENT_MOTION_PLUS_REMOVED:
347  std::cerr << "A peripheral has been removed.\n";
348  break;
349 
350  default:
351  break;
352  }
353  }
354  }
356  /* Debug section for polltime */
357  if ( 0 ) {
358  gettimeofday( &end, NULL );
359  size_t i, j;
360  if ( (i = end.tv_sec - start.tv_sec) == 0 ) j = end.tv_usec - start.tv_usec;
361  else if ( (j = end.tv_usec - start.tv_usec) < 0 ) {
362  i--;
363  j = 1000000 + j;
364  }
365  fprintf( stderr, "polltime: %lus%luu\n", i, j );
366  }
