143 void Cameras::calculateViewports( void ) {
144 
145  unsigned Width = _size.x;
146  unsigned Height = _size.y;
147 
148  /*
149  * This function is a little confusing.
150  * Ultimately, we see how many virtual cameras we have,
151  * then we compute how many "rows" and "columns" of virtual viewports
152  * we will need to accommodate that many.
153  *
154  * After those calculations, we run through a loop that goes
155  * row-by-row and then column-by-column computing the actual
156  * x,y offsets, height and widths of each viewport.
157  */
158 
159  /* Let's not try to distribute Viewports if the window system
160  hasn't been initialized yet, i.e, if resize() has not yet
161  been called. */
162  if ( !_size.x || !_size.y ) return;
163 
164  // How many cameras do we have?
165  size_t numCameras = Cameras::numCameras();
166 
167  // Let's not try to resize zero cameras.
168  if ( numCameras == 0 ) return;
169 
170  // How many virtual rows will we need to display this many?
171  size_t numRows = ceil( sqrt( numCameras ) );
172 
173  // How many columns will we need (average) in general to display this many?
174  double numCols = (double) numCameras / (double) numRows;
175  size_t numMaxCols = ceil( numCols );
176  size_t numMinCols = floor( numCols );
177 
178  // How many rows do we need to draw with MinCols?
179  // (By extension: drawMaxRows = (numRows-drawMinRows))
180  size_t drawMinRows =
181  (numCameras % numRows) ?
182  ((int) (numCameras - (numRows * numMaxCols))
183  / (int) (numMinCols - numMaxCols)) :
184  0;
185 
186  std::list< Object * >::iterator it = _list.begin();
187 
188  /* This is confusing as hell. Good luck! */
189  for ( size_t allocHeight = 0, row = 0; (row < numRows) || (it != _list.end());
190  ++row ) {
191  size_t myWidth;
192  size_t myHeight;
193  size_t allocWidth = 0;
194  size_t colsThisRow = (row < drawMinRows) ? numMinCols : numMaxCols;
195  if ( 0 )
196  fprintf( stderr, "Row: %lu; Columns this row: %lu\n", row, colsThisRow );
197 
198  for ( size_t col = 0; col < colsThisRow; ++col, ++it ) {
199  // Is this the last column? Use the remaining width.
200  if ( col + 1 == colsThisRow ) myWidth = (Width) - allocWidth;
201  else myWidth = (Width) / colsThisRow;
202  // Is this the last row? Use the remaining height.
203  if ( row + 1 == numRows ) myHeight = (Height) - allocHeight;
204  else myHeight = (Height) / numRows;
205 
206  // Tell this camera his new viewport.
207  // height looks a little goofy because we are allocating height
208  // from the top of the coordinate system and working down,
209  // so we have to take the complement.
210  obj2Cam( it )->viewport( allocWidth,
211  ((Height) - (allocHeight + myHeight)), myWidth,
212  myHeight );
213  if ( 0 )
214  fprintf( stderr, "Camera: (%lu x %lu) @ (%lu,%lu)\n", myWidth, myHeight,
215  allocWidth, ((Height) - (allocHeight + myHeight)) );
216 
217  // Increment our allocated width counter.
218  allocWidth += myWidth;
219  }
220  // Increment our allocated height counter.
221  allocHeight += ((Height) / numRows);
222  }
223 }
