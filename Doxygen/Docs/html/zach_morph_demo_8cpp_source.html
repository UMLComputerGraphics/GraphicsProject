128 void idle( void ) {

130  static Cameras *camList = Engine::instance()->cams();
131  static Scene *rootScene = Engine::instance()->rootScene();
132  Tick.Tock();
133 
134  //zach m - in order to eventually allow for user specified equations,
135  // we need to think of equations between 0 and 1.
136  // The current line "percent = timer" would be the line where
137  // we would handle a user specified equation in the form of
138  // y = x, where y is the percent and x is the timer. We may
139  // want to specify that the y should equal 1 when x equals 1,
140  // otherwise we can get a percent greater than 100.
141  static double timer = 0.0 ;
142  static bool trigger = false;
143  if(trigger == false){
144  if( (timer += 0.01) >1 ){
145  timer = 1.0;
146  trigger = true;
147  }
148  }else{
149  if( (timer -= 0.01) < 0){
150  timer = 0.0;
151  trigger = false;
152  }
153  }
154  float percent = timer;
155 
156  //if ( (timer += 0.05 ) > 360.0 ) timer = 0.0 ;
157  //float percent = ( sin(timer) + 1 ) / 2 ;
158  (*rootScene)["bottle"]->morphPercentage(percent);
159 
160  if (DEBUG_MOTION)
161  fprintf( stderr, "Time since last idle: %lu\n", Tick.Delta() );
162 
163  // Move all cameras: Apply velocity and acceleration adjustments.
164  // If no cameras are currently moving, this will do nothing ;)
165  camList->idleMotion();
166
